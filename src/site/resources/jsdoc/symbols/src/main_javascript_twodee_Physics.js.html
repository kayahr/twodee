  8 /**
  9  * Constructs a new physics model.

 11  * @constructor

 15 twodee.Physics = function()
 16 {
 17     this.velocity = new twodee.Vector();
 18     this.acceleration = new twodee.Vector();
 19     twodee.Physics.counter++;
 20 };

320  * removed at 10 seconds.

322  * @param {number} decay
323  *            The decay time in seconds

340 /**
341  * Processes the physics model for the specified node and time delta.
342  * 
343  * @param {twodee.SceneNode} node
344  *            The scene node to update
345  * @param {number} delta
346  *            The time delta 
347  */
348 twodee.Physics.prototype.process = function(node, delta)
349 {
350     var spin, transform, velocity, factor, angle, v, acceleration,
351         spinAcceleration, curVelocity, maxVelocity, minVelocity, lifetime,
352         decay, scaling;
353     
354     factor = delta / 1000;
355 
356     // Process the lifetime
357     this.lifetime = lifetime = Math.max(0, this.lifetime - factor);
358     if (!lifetime)
359     {
360         node.remove();
361         return;
362     }
363     
364     // Process the decay
365     decay = this.decay;
366     if (decay > lifetime) node.setOpacity(lifetime / decay);
367 
368     // Get the current node transform and a temporary vector
369     transform = node.getTransform();    
370     v = twodee.Physics.V;
371 
372     // Process the velocity
373     velocity = this.velocity;
374     if (!velocity.isZero())
375     {
376         angle = transform.getRotationAngle();
377         velocity.copy(v).rotate(-angle);
378         transform.translate(v.x * factor, v.y * factor);
379     }    
380 
381     // Process the acceleration
382     acceleration = this.acceleration;
383     if (!acceleration.isZero())
384     {
385         velocity.add(acceleration.copy(v).scale(factor));
386         curVelocity = velocity.getLength();
387         maxVelocity = this.maxVelocity;
388         minVelocity = this.minVelocity;
389         if (curVelocity > maxVelocity)
390             velocity.scale(maxVelocity / curVelocity);
391         else if (curVelocity < minVelocity)
392             velocity.scale(minVelocity / curVelocity);
393     }
394         
395     // Process the spinning
396     spin = this.spin;
397     if (spin) transform.rotate(spin * factor);
398     
399     // Process the scaling
400     scaling = this.scaling;
401     if (scaling != 1) transform.scale(Math.pow(scaling, factor));
402 
403     // Process the spin acceleration
404     spinAcceleration = this.spinAcceleration;
405     if (spinAcceleration)
406         this.spin = Math.max(this.minSpin, Math.min(this.maxSpin, spin +
407             spinAcceleration * factor));
408 };
